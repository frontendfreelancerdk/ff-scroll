import {AfterViewInit, Directive, ElementRef, HostListener, Input} from '@angular/core';
import {ScrollService} from './scroll.service';

@Directive({
  selector: '[ffScroll]'
})
export class ScrollDirective implements AfterViewInit {
  @Input() ffOptions: any;

  @HostListener('window:resize') onResize() {
    this.init();
  }

  constructor(private el: ElementRef, protected scrollService: ScrollService) {
  }

  ngAfterViewInit(): void {
    this.init();
  }

  init() {
    this.ffOptions.__proto__ = {
      offsetTop: 50,
      offsetLeft: 50,
      offsetRight: 50,
      offsetBottom: 50,
      beforeView: 'beforeView',
      inView: 'inView',
      afterView: 'afterView',
      animationClass: 'ff-animation',
      animationDuration: 500
    };
    this.scrollService.data = {elem: this.el.nativeElement, state: 3, options: this.ffOptions};
    this.scrollService.checkView();
  }
}
