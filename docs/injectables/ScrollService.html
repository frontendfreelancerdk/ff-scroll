import {Injectable, Renderer2, RendererFactory2} from '@angular/core';

function getOffset(elem) {
  const offsetHeight = elem.offsetHeight,
    offsetWidth = elem.offsetWidth;
  let offsetTop = 0, offsetLeft = 0;
  do {
    if (!isNaN(elem.offsetTop)) {
      offsetTop += elem.offsetTop;
    }
    if (!isNaN(elem.offsetLeft)) {
      offsetLeft += elem.offsetLeft;
    }
    elem = elem.offsetParent;
  } while (elem);
  return {
    top: offsetTop,
    right: offsetWidth + offsetLeft,
    bottom: offsetHeight + offsetTop,
    left: offsetLeft,
    height: offsetHeight,
    width: offsetWidth,
  };
}


@Injectable({
  providedIn: 'root'
})
export class ScrollService {
  _renderer: Renderer2;
  _scrollTop: number;
  _offsetTop: number;
  _offsetBottom: number;
  _windowHeight: number;
  _data: any = [];
  _state: string[];

  get data() {
    return this._data;
  }

  set data(item) {
    this._data.push(item);
  }

  constructor(private rendererFactory: RendererFactory2) {
    this._offsetBottom = 50;
    this._offsetTop = 50;

    this._renderer = rendererFactory.createRenderer(null, null);
    this.init();
    window.addEventListener('scroll', () => {
      requestAnimationFrame(() => {
        this.checkView();
      });
    });
    window.addEventListener('resize', () => {
      this.init();
    });
  }


  setCssClass(elem: HTMLElement, states, cssClass = '') {
    if (elem.className.indexOf('ff-animation') === -1) {
      for (let i = 0, len = states.length; i < len; i++) {
        this._renderer.removeClass(elem, states[i]);
      }
      this._renderer.addClass(elem, cssClass);
    }

  }

  checkView() {
    this._scrollTop = window.pageYOffset;
    this.data.forEach((item) => {
      const state = [item.options.beforeView, item.options.inView, item.options.afterView];
      console.log(state);
      let newState = this.getState(item.elem);
      if (newState !== item.state && item.elem.className.indexOf(item.options.animationClass) === -1) {
        this._renderer.addClass(item.elem, item.options.animationClass);
        this.setCssClass(item.elem, state[newState]);
        item.state = newState;
        setTimeout(() => {
          this._renderer.removeClass(item.elem, item.options.animationClass);
          newState = this.getState(item.elem);
          this.setCssClass(item.elem, state, state[newState]);
          item.state = newState;
        }, item.options.animationDuration + 20);
      }
    });
  }

  getState(elem) {
    const box = getOffset(elem);
    const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
    if (box.top > scrollTop + this._windowHeight - this._offsetBottom) {
      return 0;
    } else if (box.bottom < scrollTop + this._offsetTop) {
      return 2;
    } else {
      return 1;
    }
  }

  init() {
    this._windowHeight = document.documentElement.clientHeight;
  }
}
